# 手动请求

默认情况下，我们将会在组件挂载时，自动帮你触发绑定的请求。当然，如果你想自己控制触发请求的时机，你可以通过传入 manual 选项，来禁止挂载时的默认请求，然后通过 run 来触发请求。



import { defineComponent, ref } from 'vue';
import { useRequest } from 'vue-request';

function testService(name) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(name);
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const account = ref('John');
    const { run, data, loading } = useRequest(testService, {
      manual: true,
    });

    const handleLogin = () => {
      run(account.value);
    };
    return () => (
      <div>
        {loading.value ? 'loading...' : data.value ? `hello, ${data.value}` : ''}
        <br />
        <input v-model={[account.value]} type="text" placeholder="账号" />
        <br />
        <button onClick={handleLogin}>登录</button>
      </div>
    );
  },
});
