# 突变

当你需要在数据成功请求后，对获取回来的数据进行修改时。你可以使用 mutate 来直接修改 data 的结果

name: 

import { defineComponent, ref } from 'vue';
import { useRequest } from 'vue-request';
import Mock from 'mockjs';

function testService(name: string) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(Mock.mock('@name'));
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const name = ref('John');
    const { data, mutate, loading } = useRequest(testService);
    return () => (
      <div>
        <div>name: {loading.value ? 'loading...' : data.value}</div>
        <br />
        <div>
          <input type="text" v-model={name.value} />
          <button onClick={() => mutate(name.value)}> Edit </button>
        </div>
      </div>
    );
  },
});
