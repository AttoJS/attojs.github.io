# 高级用法

我们基于前面章节提到的基本用法做了进一步的封装，目前实现了 分页 这个场景。

# 分页扩展 New in beta.8

我们会自动帮你管理分页信息，如：当前页数、每页条数、总条数等。你只需要使用我们提供的 changeCurrent 和 changePageSize 即可轻松管理分页的数据。

注意：我们会在 params 数组第一位注入分页所需的参数。

更多可用配置请查看 分页扩展 API

# 用法

import { usePagination } from 'vue-request';

# 例子

你也可以配合 ant-design-vue 使用。这里我们提供了一个 demo，点击查看在新窗口打开

import { defineComponent } from 'vue';
import { usePagination } from 'vue-request';
import Mock from 'mockjs';
import { Loading, Pagination } from 'vant';
import 'vant/lib/pagination/style';
import 'vant/lib/loading/style';

const getData = (): string[] => Mock.mock({ 'array|5': ['@name'] }).array;
function testService() {
  return new Promise<{ total: number; result: string[] }>(resolve => {
    setTimeout(() => {
      resolve({
        total: 100,
        result: getData(),
      });
    }, 1000);
  });
}

export default defineComponent({
  name: 'App',
  setup() {
    const { data, current, total, loading, pageSize, changeCurrent } = usePagination(testService);

    return () => (
      <div>
        <div style="height: 140px">
          {loading.value ? (
            <Loading />
          ) : (
            <ul>
              {data.value?.result.map(i => (
                <li key={i}>{i}</li>
              ))}
            </ul>
          )}
        </div>
        <Pagination
          modelValue={current.value}
          total-items={total.value}
          show-page-size={pageSize.value}
          force-ellipses
          onChange={(page: number) => changeCurrent(page)}
        />
      </div>
    );
  },
});
